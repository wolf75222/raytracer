125// Refraction via loi de Snell-Descartes.
126// uv doit etre unitaire. etai_over_etat = n1/n2 (indices de refraction).
127// Decomposition en composante perpendiculaire + parallele a la normale.
128inline RT_HOSTDEV Vec3 refract(const Vec3& uv, const Vec3& n, double etai_over_etat) {
129 double cos_theta = std::fmin(dot(-uv, n), 1.0);
130 Vec3 r_out_perp = etai_over_etat * (uv + cos_theta * n);
131 Vec3 r_out_parallel = -std::sqrt(std::fabs(1.0 - r_out_perp.length_squared())) * n;
132 return r_out_perp + r_out_parallel;
133}
