37BVHAccel::BuildNode* BVHAccel::recursive_build(
38 std::vector<BuildPrimInfo>& prim_info,
39 int start, int end,
40 std::vector<std::shared_ptr<Hittable>>& src_prims,
41 int& total_nodes)
42{
43 BuildNode* node = new BuildNode();
44 total_nodes++;
45
46 // Compute bounds of all primitives in range
47 AABB bounds;
48 for (int i = start; i < end; ++i)
49 bounds = surrounding_box(bounds, prim_info[i].bbox);
50
51 int n_prims = end - start;
52
53 // Small leaf
54 if (n_prims <= 2) {
55 int first_offset = static_cast<int>(ordered_prims_.size());
56 for (int i = start; i < end; ++i)
57 ordered_prims_.push_back(src_prims[prim_info[i].prim_index]);
58 node->bbox = bounds;
59 node->first_prim_offset = first_offset;
60 node->num_primitives = n_prims;
61 return node;
62 }
63
64 // Compute centroid bounds
65 AABB centroid_bounds;
66 for (int i = start; i < end; ++i) {
67 Point3 c = prim_info[i].centroid;
68 centroid_bounds = surrounding_box(centroid_bounds, AABB(c, c));
69 }
70
71 int axis = centroid_bounds.longest_axis();
72 double axis_extent = centroid_bounds.axis_interval(axis).size();
73
74 // Degenerate: all centroids at same position
75 if (axis_extent < 1e-8) {
76 int first_offset = static_cast<int>(ordered_prims_.size());
77 for (int i = start; i < end; ++i)
78 ordered_prims_.push_back(src_prims[prim_info[i].prim_index]);
79 node->bbox = bounds;
80 node->first_prim_offset = first_offset;
81 node->num_primitives = n_prims;
82 return node;
83 }
84
85 // SAH bucket evaluation
86 constexpr int N_BUCKETS = 12;
87 struct Bucket { int count = 0; AABB bounds; };
88 Bucket buckets[N_BUCKETS];
89
90 double cmin = centroid_bounds.axis_interval(axis).min;
91 double cmax = centroid_bounds.axis_interval(axis).max;
92
93 for (int i = start; i < end; ++i) {
94 int b = static_cast<int>(N_BUCKETS *
95 ((prim_info[i].centroid[axis] - cmin) / (cmax - cmin)));
96 if (b >= N_BUCKETS) b = N_BUCKETS - 1;
97 buckets[b].count++;
98 buckets[b].bounds = surrounding_box(buckets[b].bounds, prim_info[i].bbox);
99 }
100
101 // Evaluate cost for each split position
102 constexpr double T_TRAV = 0.125;
103 constexpr double T_ISECT = 1.0;
104 double parent_area = bounds.surface_area();
105 double best_cost = std::numeric_limits<double>::infinity();
106 int best_split = -1;
107
108 // Forward sweep: accumulate left side
109 double cost_left_area[N_BUCKETS - 1];
110 int cost_left_count[N_BUCKETS - 1];
111 {
112 AABB accum;
113 int count = 0;
114 for (int i = 0; i < N_BUCKETS - 1; ++i) {
115 accum = surrounding_box(accum, buckets[i].bounds);
116 count += buckets[i].count;
117 cost_left_area[i] = accum.surface_area();
118 cost_left_count[i] = count;
119 }
120 }
121 // Backward sweep: accumulate right side and find best
122 {
123 AABB accum;
124 int count = 0;
125 for (int i = N_BUCKETS - 1; i >= 1; --i) {
126 accum = surrounding_box(accum, buckets[i].bounds);
127 count += buckets[i].count;
128 double cost = T_TRAV + T_ISECT *
129 (cost_left_count[i-1] * cost_left_area[i-1] +
130 count * accum.surface_area()) / parent_area;
131 if (cost < best_cost) {
132 best_cost = cost;
133 best_split = i;
134 }
135 }
136 }
137
138 // Compare with leaf cost
139 double leaf_cost = T_ISECT * n_prims;
140 if (best_cost >= leaf_cost && n_prims <= 16) {
141 // Leaf is cheaper
142 int first_offset = static_cast<int>(ordered_prims_.size());
143 for (int i = start; i < end; ++i)
144 ordered_prims_.push_back(src_prims[prim_info[i].prim_index]);
145 node->bbox = bounds;
146 node->first_prim_offset = first_offset;
147 node->num_primitives = n_prims;
148 return node;
149 }
150
151 // Partition primitives at best split
152 auto mid_it = std::partition(
153 prim_info.begin() + start, prim_info.begin() + end,
154 [&](const BuildPrimInfo& pi) {
155 int b = static_cast<int>(N_BUCKETS *
156 ((pi.centroid[axis] - cmin) / (cmax - cmin)));
157 if (b >= N_BUCKETS) b = N_BUCKETS - 1;
158 return b < best_split;
159 });
160 int mid = static_cast<int>(std::distance(prim_info.begin(), mid_it));
161
162 if (mid == start || mid == end)
163 mid = start + n_prims / 2;
164
165 // Recurse
166 node->children[0] = recursive_build(prim_info, start, mid, src_prims, total_nodes);
167 node->children[1] = recursive_build(prim_info, mid, end, src_prims, total_nodes);
168 node->bbox = surrounding_box(node->children[0]->bbox, node->children[1]->bbox);
169 node->split_axis = axis;
170 node->num_primitives = 0;
171 return node;
172}

175 FlatBVHNode& flat = nodes_[offset];

203bool BVHAccel::hit(const Ray& r, Interval ray_t, HitRecord& rec) const {
204 if (nodes_.empty()) return false;
205
206 bool hit_anything = false;
207 double closest = ray_t.max;
208
209 // Iterative traversal with explicit stack
210 constexpr int MAX_STACK = 64;
211 int stack[MAX_STACK];
212 int stack_ptr = 0;
213 stack[stack_ptr++] = 0;
214
215 while (stack_ptr > 0) {
216 int node_idx = stack[--stack_ptr];
217 const FlatBVHNode& node = nodes_[node_idx];
218
219 if (!node.bbox.hit(r, ray_t.min, closest))
220 continue;
221
222 if (node.num_primitives > 0) {
223 // Leaf: test all primitives
224 for (uint16_t i = 0; i < node.num_primitives; ++i) {
225 if (ordered_prims_[node.primitives_offset + i]->hit(
226 r, Interval(ray_t.min, closest), rec)) {
227 hit_anything = true;
228 closest = rec.t;
229 }
230 }
231 } else {
232 // Interior: push children in front-to-back order
233 // Left child = node_idx + 1, right child = second_child_offset
234 int near_child = node_idx + 1;
235 int far_child = node.second_child_offset;
236
237 // If ray direction is negative along split axis, swap
238 if (r.dir_sign[node.split_axis]) {
239 int tmp = near_child; near_child = far_child; far_child = tmp;
240 }
241
242 // Push far first (popped last = traversed last)
243 stack[stack_ptr++] = far_child;
244 stack[stack_ptr++] = near_child;
245 }
246 }
247 return hit_anything;
248}

284 < b->bounding_box().axis_interval(axis).min;
285 });
286
287 size_t mid = start + object_span / 2;
288 left_ = std::make_shared<BVHNode>(objects, start, mid);
289 right_ = std::make_shared<BVHNode>(objects, mid, end);
