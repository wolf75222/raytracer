28// Inline scatter to avoid virtual dispatch in the hot path
29static inline bool scatter_inline(const Material* mat, const Ray& r_in,
30 const HitRecord& rec,
31 Color& attenuation, Ray& scattered) {
32 switch (mat->type()) {
33 case Material::Type::Lambertian: {
34 const auto* lam = static_cast<const Lambertian*>(mat);
35 Vec3 scatter_dir = rec.normal + random_unit_vector();
36 if (scatter_dir.near_zero()) scatter_dir = rec.normal;
37 scattered = Ray(rec.p, scatter_dir);
38 attenuation = lam->albedo();
39 return true;
40 }
41 case Material::Type::Metal: {
42 const auto* met = static_cast<const Metal*>(mat);
43 Vec3 reflected = reflect(unit_vector(r_in.direction()), rec.normal);
44 reflected = reflected + met->fuzz() * random_in_unit_sphere();
45 scattered = Ray(rec.p, reflected);
46 attenuation = met->albedo();
47 return (dot(scattered.direction(), rec.normal) > 0.0);
48 }
49 case Material::Type::Dielectric: {
50 const auto* die = static_cast<const Dielectric*>(mat);
51 attenuation = Color(1.0, 1.0, 1.0);
52 double ri_val = die->refraction_index();
53 double ri = rec.front_face ? (1.0 / ri_val) : ri_val;
54 Vec3 unit_dir = unit_vector(r_in.direction());
55 double cos_theta = std::fmin(dot(-unit_dir, rec.normal), 1.0);
56 double sin_theta = std::sqrt(1.0 - cos_theta * cos_theta);
57 bool cannot_refract = ri * sin_theta > 1.0;
58 Vec3 direction;
59 if (cannot_refract || Dielectric::reflectance(cos_theta, ri) > random_double())
60 direction = reflect(unit_dir, rec.normal);
61 else
62 direction = refract(unit_dir, rec.normal, ri);
63 scattered = Ray(rec.p, direction);
64 return true;
65 }
66 case Material::Type::Checker: {
67 // Devirtualise comme Lambertian mais avec couleur alternee
68 return mat->scatter(r_in, rec, attenuation, scattered);
69 }
70 default:
71 return mat->scatter(r_in, rec, attenuation, scattered);
72 }
73}

130void CpuRenderer::render_threaded(int num_threads) {
131 if (num_threads <= 0)
132 num_threads = static_cast<int>(std::thread::hardware_concurrency());
133 if (num_threads <= 0)
134 num_threads = 4;
135
136 const int width = image_.width();
137 const int height = image_.height();
138 constexpr int TILE_SIZE = 16;
139
140 const int tiles_x = (width + TILE_SIZE - 1) / TILE_SIZE;
141 const int tiles_y = (height + TILE_SIZE - 1) / TILE_SIZE;
142 const int total_tiles = tiles_x * tiles_y;
143
144 std::atomic<int> next_tile(0);
145 std::atomic<int> completed_tiles(0);
146
147 auto worker = [&]() {
148 while (true) {
149 int tile_idx = next_tile.fetch_add(1);
150 if (tile_idx >= total_tiles)
151 break;
152
153 int tx = tile_idx % tiles_x;
154 int ty = tile_idx / tiles_x;
155 int x0 = tx * TILE_SIZE;
156 int y0 = ty * TILE_SIZE;
157 int x1 = std::min(x0 + TILE_SIZE, width);
158 int y1 = std::min(y0 + TILE_SIZE, height);
159
160 render_tile(x0, y0, x1, y1);
161
162 int done = completed_tiles.fetch_add(1) + 1;
163 if (done % 10 == 0 || done == total_tiles) {
164 std::cerr << "\rTiles: " << done << "/" << total_tiles
165 << " (" << (100 * done / total_tiles) << "%)" << std::flush;
166 }
167 }
168 };
169
170 std::cerr << "Rendering with " << num_threads << " threads, "
171 << total_tiles << " tiles (" << TILE_SIZE << "x" << TILE_SIZE << ")...\n";
172
173 std::vector<std::thread> threads;
174 threads.reserve(num_threads);
175 for (int i = 0; i < num_threads; ++i)
176 threads.emplace_back(worker);
177
178 for (auto& t : threads)
179 t.join();
180
181 std::cerr << "\rDone. \n";
182}

184// CS:APP 5.5 + 5.6: Iterative ray_color with devirtualized scatter
185// - No recursion (eliminates stack frame per bounce)
186// - Inline scatter avoids vtable lookup per bounce
187// - Local variables eliminate repeated memory dereferences
188Color CpuRenderer::ray_color(const Ray& r, const Hittable& world, int depth) const {
189 Color throughput(1, 1, 1);
190 Ray current_ray = r;
191 constexpr double T_MIN = 0.001;
192 constexpr double T_MAX = 1e30; // CS:APP 5.6: avoid std::numeric_limits call
193
194 for (int d = 0; d < depth; ++d) {
195 HitRecord rec;
196 if (!world.hit(current_ray, Interval(T_MIN, T_MAX), rec)) {
197 // Sky: CS:APP 5.6 - compute with local values
198 const Vec3& dir = current_ray.direction();
199 // Inline unit_vector Y component to avoid full normalization
200 // We only need the Y component for the sky gradient
201 double len_sq = dir.length_squared();
202 double inv_len = 1.0 / std::sqrt(len_sq);
203 double unit_y = dir.y() * inv_len;
204 double a = 0.5 * (unit_y + 1.0);
205 // Sky color: lerp white->blue
206 Color sky(1.0 - 0.5 * a, 1.0 - 0.3 * a, 1.0);
207 return throughput * sky;
208 }
209
210 // CS:APP 5.5: Devirtualized scatter - switch on type tag, no vtable
211 const Material* mat = rec.material;
212 if (!mat) break;
213
214 // Area light : si emissif, retourner la lumiere emise
215 if (mat->type() == Material::Type::Emissive) {
216 const auto* em = static_cast<const Emissive*>(mat);
217 return throughput * em->emitted();
218 }
219
220 Ray scattered;
221 Color attenuation;
222 if (!scatter_inline(mat, current_ray, rec, attenuation, scattered))
223 break; // absorbed
224
225 // CS:APP 5.6: Multiply throughput components directly
226 throughput = throughput * attenuation;
227
228 // Early termination + Russian Roulette (RTIOW CUDA article Opt#4+5)
229 double max_comp = std::fmax(throughput.x(), std::fmax(throughput.y(), throughput.z()));
230 if (max_comp < 0.001)
231 break;
232 if (d > 3) {
233 if (random_double() > max_comp)
234 break;
235 throughput = throughput / max_comp;
236 }
237
238 current_ray = scattered;
239 }
240
241 return Color(0, 0, 0);
242}
