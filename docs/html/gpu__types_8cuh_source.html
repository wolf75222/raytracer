77struct GAABB {
78 GVec3 mn, mx; // min, max
79
80 __host__ __device__ GAABB() : mn(1e30f, 1e30f, 1e30f), mx(-1e30f, -1e30f, -1e30f) {}

83 __host__ __device__ bool hit(const GRay& r, float t_min, float t_max) const {
84 // FMA-based slab test (Opt#7 level 4 from RTIOW CUDA article)
85 // fmaf(invDir, bound, origMulInv) = invDir*bound + (-orig*invDir) = (bound-orig)*invDir
86 float tx0 = fmaf(r.inv_dir.x, mn.x, r.orig_mul_inv.x);
87 float tx1 = fmaf(r.inv_dir.x, mx.x, r.orig_mul_inv.x);
88 float ty0 = fmaf(r.inv_dir.y, mn.y, r.orig_mul_inv.y);
89 float ty1 = fmaf(r.inv_dir.y, mx.y, r.orig_mul_inv.y);
90 float tz0 = fmaf(r.inv_dir.z, mn.z, r.orig_mul_inv.z);
91 float tz1 = fmaf(r.inv_dir.z, mx.z, r.orig_mul_inv.z);
92
93 float tEnter = fmaxf(fmaxf(fminf(tx0,tx1), fminf(ty0,ty1)), fmaxf(fminf(tz0,tz1), t_min));
94 float tExit = fminf(fminf(fmaxf(tx0,tx1), fmaxf(ty0,ty1)), fminf(fmaxf(tz0,tz1), t_max));
95 return tEnter <= tExit;
96 }
97};
