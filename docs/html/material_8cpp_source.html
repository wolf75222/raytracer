55bool Dielectric::scatter(const Ray& r_in, const HitRecord& rec,
56 Color& attenuation, Ray& scattered) const {
57 attenuation = Color(1.0, 1.0, 1.0);
58 double ri = rec.front_face ? (1.0 / refraction_index_) : refraction_index_;
59
60 Vec3 unit_dir = unit_vector(r_in.direction());
61 double cos_theta = std::fmin(dot(-unit_dir, rec.normal), 1.0);
62 double sin_theta = std::sqrt(1.0 - cos_theta * cos_theta);
63
64 bool cannot_refract = ri * sin_theta > 1.0;
65 Vec3 direction;
66

76double Dielectric::reflectance(double cosine, double refraction_index) {
77 // Schlick's approximation
78 double r0 = (1.0 - refraction_index) / (1.0 + refraction_index);
79 r0 = r0 * r0;
80 double omc = 1.0 - cosine;
81 double omc2 = omc * omc;
82 return r0 + (1.0 - r0) * omc2 * omc2 * omc;
83}
