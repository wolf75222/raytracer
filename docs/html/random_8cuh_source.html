11// Mersenne Twister 19937 : periode 2^19937-1, qualite statistique suffisante
12// pour le path tracing. Seed via random_device (entropie hardware si disponible).
13inline double random_double() {
14 thread_local std::mt19937 gen(std::random_device{}());
15 thread_local std::uniform_real_distribution<double> dist(0.0, 1.0);
16 return dist(gen);
17}

19inline double random_double(double min, double max) {
20 return min + (max - min) * random_double();

31// Rejection sampling dans le cube [-1,1]^3 : ~52% d'acceptance.
32// Plus simple et aussi rapide que Box-Muller pour 3 composantes.
33inline Vec3 random_in_unit_sphere() {
34 while (true) {
35 Vec3 p = random_vec3(-1.0, 1.0);
36 if (p.length_squared() < 1.0)
37 return p;
38 }
39}

41// Direction aleatoire uniforme sur la sphere unite :
42// normaliser un point uniforme dans la boule donne une distribution
43// uniforme sur la surface (distribution de Lambert pour le diffus).
44inline Vec3 random_unit_vector() {
45 return unit_vector(random_in_unit_sphere());

48// Hemisphere cosinus-pondere implicite : si le vecteur pointe
49// du mauvais cote de la normale, on le retourne.
50inline Vec3 random_on_hemisphere(const Vec3& normal) {
51 Vec3 on_sphere = random_unit_vector();
52 if (dot(on_sphere, normal) > 0.0)
53 return on_sphere;
54 return -on_sphere;
55}
