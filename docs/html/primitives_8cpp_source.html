11bool Sphere::hit(const Ray& r, Interval ray_t, HitRecord& rec) const {
12 Vec3 oc = r.origin() - center_;
13 double a = r.direction().length_squared();
14 double half_b = dot(oc, r.direction());
15 double c = oc.length_squared() - radius_ * radius_;
16
17 double discriminant = half_b * half_b - a * c;
18 if (discriminant < 0.0)
19 return false;
20
21 double sqrtd = std::sqrt(discriminant);
22
23 // Find nearest root in the acceptable range
24 double inv_a = 1.0 / a;
25 double root = (-half_b - sqrtd) * inv_a;
26 if (!ray_t.surrounds(root)) {
27 root = (-half_b + sqrtd) * inv_a;
28 if (!ray_t.surrounds(root))
29 return false;
30 }
31
32 rec.t = root;
33 rec.p = r.at(rec.t);
34 Vec3 outward_normal = (rec.p - center_) * (1.0 / radius_);
35 rec.set_face_normal(r, outward_normal);
36 rec.material = material_;
37
38 return true;
39}

73AABB Plane::bounding_box() const {
74 // Infinite plane: use a very large bounding box
75 // Slightly thicken along normal to avoid zero-volume AABB
76 constexpr double big = 1e4;
77 constexpr double eps = 0.0001;
78 Point3 lo = point_ - Vec3(big, big, big);

98bool Cylinder::hit(const Ray& r, Interval ray_t, HitRecord& rec) const {
99 // Intersection rayon-cylindre infini (axe Y)
100 double ox = r.origin().x() - base_.x();
101 double oz = r.origin().z() - base_.z();
102 double dx = r.direction().x();
103 double dz = r.direction().z();
104
105 double a = dx*dx + dz*dz;
106 double half_b = ox*dx + oz*dz;
107 double c = ox*ox + oz*oz - radius_*radius_;
108
109 double disc = half_b*half_b - a*c;
110 if (disc < 0.0) return false;
111
112 double sqrtd = std::sqrt(disc);
113 double root = (-half_b - sqrtd) / a;
114
115 // Tester les deux racines
116 for (int i = 0; i < 2; i++) {
117 if (i == 1) root = (-half_b + sqrtd) / a;
118 if (!ray_t.surrounds(root)) continue;
119
120 double y = r.origin().y() + root * r.direction().y();
121 if (y < base_.y() || y > base_.y() + height_) continue;
122
123 rec.t = root;
124 rec.p = r.at(root);
125 // Normale du cylindre : direction radiale (pas de composante Y)
126 Vec3 outward_normal = Vec3(rec.p.x() - base_.x(), 0, rec.p.z() - base_.z()) / radius_;
127 rec.set_face_normal(r, outward_normal);
128 rec.material = material_;
129 return true;
130 }
131 return false;
132}

150bool Triangle::hit(const Ray& r, Interval ray_t, HitRecord& rec) const {
151 Vec3 edge1 = v1_ - v0_;
152 Vec3 edge2 = v2_ - v0_;
153 Vec3 h = cross(r.direction(), edge2);
154 double a = dot(edge1, h);
155
156 if (std::fabs(a) < 1e-8)
157 return false; // rayon parallele au triangle
158
159 double f = 1.0 / a;
160 Vec3 s = r.origin() - v0_;
161 double u = f * dot(s, h);
162 if (u < 0.0 || u > 1.0)
163 return false;
164
165 Vec3 q = cross(s, edge1);
166 double v = f * dot(r.direction(), q);
167 if (v < 0.0 || u + v > 1.0)
168 return false;
169
170 double t = f * dot(edge2, q);
171 if (!ray_t.surrounds(t))
172 return false;
173
174 rec.t = t;
175 rec.p = r.at(t);
176 rec.set_face_normal(r, normal_);
177 rec.material = material_;
178 return true;
179}
