22void hybrid_render_scene(const Scene& scene, ImageBuffer& image,
23 float gpu_fraction,
24 float& cpu_time_ms, float& gpu_time_ms) {
25 int w = image.width();
26 int h = image.height();
27 constexpr int TILE_SIZE = 32;
28
29 int tiles_x = (w + TILE_SIZE - 1) / TILE_SIZE;
30 int tiles_y = (h + TILE_SIZE - 1) / TILE_SIZE;
31 int total_tiles = tiles_x * tiles_y;
32
33 // Repartition : GPU prend les premieres tuiles, CPU le reste
34 int gpu_tiles = (int)(total_tiles * gpu_fraction);
35 if (gpu_tiles < 0) gpu_tiles = 0;
36 if (gpu_tiles > total_tiles) gpu_tiles = total_tiles;
37 int cpu_start_tile = gpu_tiles;
38
39 // Calcul des lignes GPU (tuiles completes depuis le haut)
40 int gpu_rows = 0;
41 if (gpu_tiles > 0) {
42 int gpu_tile_rows = (gpu_tiles + tiles_x - 1) / tiles_x;
43 gpu_rows = gpu_tile_rows * TILE_SIZE;
44 if (gpu_rows > h) gpu_rows = h;
45 gpu_tiles = gpu_tile_rows * tiles_x; // ajuster au nombre reel
46 cpu_start_tile = gpu_tiles;
47 }
48
49 fprintf(stderr, "Hybrid: GPU %d tuiles (lignes 0-%d), CPU %d tuiles (lignes %d-%d), total %d\n",
50 gpu_tiles, gpu_rows, total_tiles - cpu_start_tile, gpu_rows, h, total_tiles);
51
52 // --- GPU thread : rend la partie haute ---
53 std::thread gpu_thread;
54 float gpu_kernel_ms = 0;
55
56 if (gpu_rows > 0) {
57 gpu_thread = std::thread([&]() {
58 auto t0 = std::chrono::high_resolution_clock::now();
59
60 // GPU rend toute l'image (plus simple que de modifier le kernel)
61 // On ne copie que les lignes GPU dans l'ImageBuffer
62 float km = 0;
63 gpu_render_scene(scene, image, km);
64 gpu_kernel_ms = km;
65
66 auto t1 = std::chrono::high_resolution_clock::now();
67 gpu_time_ms = (float)std::chrono::duration<double, std::milli>(t1 - t0).count();
68 });
69 }
70
71 // --- CPU : rend les tuiles restantes avec work stealing ---
72 if (cpu_start_tile < total_tiles) {
73 auto t0 = std::chrono::high_resolution_clock::now();
74
75 CpuRenderer cpu_renderer(scene, image);
76 cpu_renderer.enable_bvh();
77
78 int num_threads = (int)std::thread::hardware_concurrency();
79 if (num_threads <= 0) num_threads = 4;
80
81 int cpu_total = total_tiles - cpu_start_tile;
82 std::atomic<int> next_tile(0);
83
84 auto worker = [&]() {
85 while (true) {
86 int local_idx = next_tile.fetch_add(1);
87 if (local_idx >= cpu_total) break;
88
89 int tile_idx = cpu_start_tile + local_idx;
90 int tx = tile_idx % tiles_x;
91 int ty = tile_idx / tiles_x;
92 int x0 = tx * TILE_SIZE;
93 int y0 = ty * TILE_SIZE;
94 int x1 = std::min(x0 + TILE_SIZE, w);
95 int y1 = std::min(y0 + TILE_SIZE, h);
96
97 cpu_renderer.render_tile(x0, y0, x1, y1);
98 }
99 };
100
101 std::vector<std::thread> threads;
102 for (int i = 0; i < num_threads; i++)
103 threads.emplace_back(worker);
104 for (auto& t : threads)
105 t.join();
106
107 auto t1 = std::chrono::high_resolution_clock::now();
108 cpu_time_ms = (float)std::chrono::duration<double, std::milli>(t1 - t0).count();
109 }
110
111 if (gpu_thread.joinable())
112 gpu_thread.join();
113
114 fprintf(stderr, "Hybrid: CPU %.1fms, GPU kernel %.1fms (total GPU %.1fms)\n",
115 cpu_time_ms, gpu_kernel_ms, gpu_time_ms);
116}
