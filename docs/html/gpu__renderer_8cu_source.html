47__device__ bool intersect_sphere(const GSphere& s, const GRay& r, float t_min, float t_max, GHitRecord& rec) {
48 GVec3 oc = r.orig - s.center;
49 float a = dot(r.dir, r.dir);
50 float half_b = dot(oc, r.dir);
51 float c = oc.length_squared() - s.radius * s.radius;
52 float disc = half_b * half_b - a * c;
53 if (disc < 0.0f) return false;
54 float sqrtd = sqrtf(disc);
55 float inv_a = __fdividef(1.0f, a); // fast HW reciprocal, avoids full division
56 float root = (-half_b - sqrtd) * inv_a;
57 if (root < t_min || root > t_max) {
58 root = (-half_b + sqrtd) * inv_a;
59 if (root < t_min || root > t_max) return false;
60 }
61 rec.t = root;
62 rec.p = r.at(root);
63 float inv_r = __fdividef(1.0f, s.radius);
64 rec.set_face_normal(r, (rec.p - s.center) * inv_r);
65 rec.material_id = s.material_id;
66 return true;
67}

69// SoA layout: coalesced loads when warp threads test consecutive spheres
70__device__ bool intersect_sphere_soa(const GSphereSoA& soa, int idx,
71 const GRay& r, float t_min, float t_max, GHitRecord& rec) {
72 float sx = soa.cx[idx], sy = soa.cy[idx], sz = soa.cz[idx], sr = soa.r[idx];
73 GVec3 oc(r.orig.x - sx, r.orig.y - sy, r.orig.z - sz);
74 float a = dot(r.dir, r.dir);
75 float half_b = dot(oc, r.dir);
76 float c = oc.length_squared() - sr * sr;
77 float disc = half_b * half_b - a * c;
78 if (disc < 0.0f) return false;
79 float sqrtd = sqrtf(disc);
80 float inv_a = __fdividef(1.0f, a);
81 float root = (-half_b - sqrtd) * inv_a;
82 if (root < t_min || root > t_max) {
83 root = (-half_b + sqrtd) * inv_a;
84 if (root < t_min || root > t_max) return false;
85 }
86 rec.t = root;
87 rec.p = r.at(root);
88 float inv_r = __fdividef(1.0f, sr);
89 rec.set_face_normal(r, GVec3(rec.p.x - sx, rec.p.y - sy, rec.p.z - sz) * inv_r);
90 rec.material_id = soa.mat[idx];
91 return true;
92}

94__device__ bool intersect_plane(const GPlane& pl, const GRay& r, float t_min, float t_max, GHitRecord& rec) {
95 float denom = dot(r.dir, pl.normal);
96 if (fabsf(denom) < 1e-6f) return false;
97 float t = dot(pl.point - r.orig, pl.normal) * __fdividef(1.0f, denom);
98 if (t < t_min || t > t_max) return false;
99 rec.t = t;
100 rec.p = r.at(t);
101 rec.set_face_normal(r, pl.normal);
102 rec.material_id = pl.material_id;
103 return true;
104}

106// ============================================================
107// BVH traversal iteratif (stack fixe, front-to-back)
108// ============================================================
109// Top BVH nodes in shared mem: accessed by nearly all rays in the block
110#define BVH_SHARED_NODES 7 // 3 premiers niveaux = 1+2+4 = 7 nodes

112__device__ bool trace_bvh(const GBVHNode* __restrict__ nodes, int num_nodes,
113 const GSphereSoA& soa,
114 const GPlane* __restrict__ planes,
115 const GRay& r, float t_min, float t_max,
116 GHitRecord& rec,
117 const GBVHNode* __restrict__ sh_bvh, int sh_count) {
118 bool hit_anything = false;
119 float closest = t_max;
120 int stack[16];
121 int sp = 0;
122 stack[sp++] = 0;
123
124 while (sp > 0) {
125 int idx = stack[--sp];
126 const GBVHNode& node = (idx < sh_count) ? sh_bvh[idx] : nodes[idx];
127
128 if (!node.bbox.hit(r, t_min, closest))
129 continue;
130
131 if (node.num_prims > 0) {
132 #pragma unroll 4
133 for (uint16_t i = 0; i < node.num_prims; i++) {
134 GHitRecord tmp;
135 // SoA: each float[] is contiguous -> coalesced reads
136 if (intersect_sphere_soa(soa, node.data + i, r, t_min, closest, tmp)) {
137 hit_anything = true; closest = tmp.t; rec = tmp;
138 }
139 }
140 } else {
141 int near_child = idx + 1;
142 int far_child = node.data;
143 float dir_comp = (node.split_axis == 0) ? r.dir.x :
144 (node.split_axis == 1) ? r.dir.y : r.dir.z;
145 if (dir_comp < 0.0f) { int tmp = near_child; near_child = far_child; far_child = tmp; }
146 stack[sp++] = far_child;
147 stack[sp++] = near_child;
148 }
149 }
150 return hit_anything;
151}

153// Brute force fallback
154__device__ bool trace_brute(const GSphere* spheres, int num_spheres,
155 const GPlane* planes, int num_planes,
156 const GRay& r, float t_min, float t_max,
157 GHitRecord& rec) {
158 bool hit = false;
159 float closest = t_max;
160 GHitRecord tmp;
161 for (int i = 0; i < num_spheres; i++) {
162 if (intersect_sphere(spheres[i], r, t_min, closest, tmp)) {
163 hit = true; closest = tmp.t; rec = tmp;
164 }
165 }
166 for (int i = 0; i < num_planes; i++) {
167 if (intersect_plane(planes[i], r, t_min, closest, tmp)) {
168 hit = true; closest = tmp.t; rec = tmp;
169 }
170 }
171 return hit;
172}

174// ============================================================
175// Material scatter
176// ============================================================
177__device__ bool scatter_mat(const GMaterial& mat, const GRay& r_in, const GHitRecord& rec,
178 GVec3& attenuation, GRay& scattered, GRng& rng) {
179 switch (mat.type) {
180 case GMAT_LAMBERTIAN: {
181 GVec3 dir = rec.normal + random_unit_vector(rng);
182 if (dir.near_zero()) dir = rec.normal;
183 scattered = GRay(rec.p, dir);
184 attenuation = mat.albedo;
185 return true;
186 }
187 case GMAT_METAL: {
188 GVec3 reflected = reflect(normalize(r_in.dir), rec.normal);
189 reflected = reflected + mat.param * random_in_unit_sphere(rng);
190 scattered = GRay(rec.p, reflected);
191 attenuation = mat.albedo;
192 return dot(scattered.dir, rec.normal) > 0.0f;
193 }
194 case GMAT_DIELECTRIC: {
195 attenuation = mat.albedo; // tinted glass support
196 float ri = rec.front_face ? (1.0f / mat.param) : mat.param;
197 GVec3 unit_dir = normalize(r_in.dir);
198 float cos_theta = fminf(dot(-unit_dir, rec.normal), 1.0f);
199 float sin_theta = sqrtf(1.0f - cos_theta * cos_theta);
200 float r0 = (1.0f - ri) / (1.0f + ri); r0 = r0 * r0;
201 // manual pow5: avoids expensive powf (exp+log chain)
202 float omc = 1.0f - cos_theta;
203 float omc2 = omc * omc;
204 float schlick = r0 + (1.0f - r0) * omc2 * omc2 * omc;
205 GVec3 direction;
206 if (ri * sin_theta > 1.0f || schlick > grng_float(rng))
207 direction = reflect(unit_dir, rec.normal);
208 else
209 direction = refract(unit_dir, rec.normal, ri);
210 scattered = GRay(rec.p, direction);
211 return true;
212 }
213 }
214 return false;
215}

217// ============================================================
218// Ray color iteratif
219// ============================================================
220__device__ GVec3 ray_color(GRay r,
221 const GBVHNode* bvh, int num_bvh,
222 const GSphereSoA& soa,
223 const GSphere* spheres, int num_spheres,
224 const GPlane* planes, int num_planes,
225 const GMaterial* materials,
226 int max_depth, GRng& rng,
227 const GBVHNode* sh_bvh, int sh_count) {
228 GVec3 throughput(1.0f, 1.0f, 1.0f);
229
230 for (int d = 0; d < max_depth; d++) {
231 GHitRecord rec;
232 bool hit;
233 if (num_bvh > 0) {
234 // BVH pour les spheres + brute force pour les plans (peu nombreux)
235 float closest = 1e30f;
236 hit = trace_bvh(bvh, num_bvh, soa, planes, r, 0.001f, closest, rec, sh_bvh, sh_count);
237 if (hit) closest = rec.t;
238 // Tester les plans en brute force
239 GHitRecord tmp;
240 for (int i = 0; i < num_planes; i++) {
241 if (intersect_plane(planes[i], r, 0.001f, closest, tmp)) {
242 hit = true; closest = tmp.t; rec = tmp;
243 }
244 }
245 } else {
246 hit = trace_brute(spheres, num_spheres, planes, num_planes, r, 0.001f, 1e30f, rec);
247 }
248 if (!hit) {
249 GVec3 unit_dir = normalize(r.dir);
250 float a = 0.5f * (unit_dir.y + 1.0f);
251 GVec3 sky(1.0f - 0.5f*a, 1.0f - 0.3f*a, 1.0f);
252 return throughput * sky;
253 }
254 const GMaterial& mat = materials[rec.material_id];
255 if (mat.type == GMAT_EMISSIVE)
256 return throughput * mat.albedo * mat.param;
257
258 GVec3 attenuation;
259 GRay scattered;
260 if (!scatter_mat(mat, r, rec, attenuation, scattered, rng))
261 return GVec3(0, 0, 0);
262 throughput = throughput * attenuation;
263
264 // Russian roulette: unbiased path termination after depth 3
265 float max_comp = fmaxf(throughput.x, fmaxf(throughput.y, throughput.z));
266 if (max_comp < 0.001f)
267 break;
268
269 if (d > 3) {
270 if (grng_float(rng) > max_comp)
271 break;
272 throughput = throughput * (1.0f / max_comp);
273 }
274
275 r = scattered;
276 }
277 return GVec3(0, 0, 0);
278}

288__global__ void render_kernel(float* __restrict__ framebuffer, int width, int height,
289 const GBVHNode* __restrict__ bvh, int num_bvh,
290 const GSphere* __restrict__ spheres, int num_spheres,
291 const GPlane* __restrict__ planes, int num_planes,
292 const GMaterial* __restrict__ materials,
293 const float* __restrict__ soa_cx, const float* __restrict__ soa_cy,
294 const float* __restrict__ soa_cz, const float* __restrict__ soa_r,
295 const int* __restrict__ soa_mat) {
296 // Cooperative load of top BVH nodes into shared mem
297 __shared__ GBVHNode sh_bvh[BVH_SHARED_NODES];
298 int tid_flat = threadIdx.y * blockDim.x + threadIdx.x;
299 if (tid_flat < BVH_SHARED_NODES && tid_flat < num_bvh) {
300 sh_bvh[tid_flat] = bvh[tid_flat];
301 }
302 __syncthreads(); // Attendre que tous les top nodes soient en shared mem
303
304 int sh_count = (num_bvh < BVH_SHARED_NODES) ? num_bvh : BVH_SHARED_NODES;
305
306 int x = blockIdx.x * blockDim.x + threadIdx.x;
307 int y = blockIdx.y * blockDim.y + threadIdx.y;
308 if (x >= width || y >= height) return;
309
310 GRng rng;
311 grng_init(rng, y * width + x, 42);
312
313 GVec3 pixel_color(0, 0, 0);
314 int spp = d_camera.samples_per_pixel;
315 float inv_spp = 1.0f / (float)spp;
316
317 for (int s = 0; s < spp; s++) {
318 float u_off = grng_float(rng) - 0.5f;
319 float v_off = grng_float(rng) - 0.5f;
320 GVec3 pixel_sample = d_camera.pixel00_loc
321 + ((float)x + u_off) * d_camera.pixel_delta_u
322 + ((float)y + v_off) * d_camera.pixel_delta_v;
323 GVec3 ray_origin = d_camera.origin;
324 if (d_camera.defocus_angle > 0.0f) {
325 GVec3 pd = random_in_unit_disk(rng);
326 ray_origin = d_camera.origin + pd.x * d_camera.defocus_disk_u + pd.y * d_camera.defocus_disk_v;
327 }
328 GRay r(ray_origin, pixel_sample - ray_origin);
329 GSphereSoA soa;
330 soa.cx = const_cast<float*>(soa_cx); soa.cy = const_cast<float*>(soa_cy);
331 soa.cz = const_cast<float*>(soa_cz); soa.r = const_cast<float*>(soa_r);
332 soa.mat = const_cast<int*>(soa_mat); soa.count = num_spheres;
333 pixel_color += ray_color(r, bvh, num_bvh, soa, spheres, num_spheres, planes, num_planes,
334 materials, d_camera.max_depth, rng, sh_bvh, sh_count);
335 }
336
337 pixel_color *= inv_spp;
338
339 // Planar layout (RRR...GGG...BBB): stride-1 writes for coalescing
340 int pixel_idx = y * width + x;
341 int total_pixels = width * height;
342 // Linear HDR output, gamma/tonemap done later
343 framebuffer[pixel_idx] = fmaxf(pixel_color.x, 0.0f);
344 framebuffer[pixel_idx + total_pixels] = fmaxf(pixel_color.y, 0.0f);
345 framebuffer[pixel_idx + total_pixels*2] = fmaxf(pixel_color.z, 0.0f);
346}

348// ============================================================
349// Tonemap GPU kernel : lineaire -> ACES -> gamma -> ARGB 8-bit
350// Evite le transfert float D->H + tonemap CPU
351// ============================================================
352__device__ float aces_tonemap_gpu(float x) {
353 float a = x * (2.51f * x + 0.03f);
354 float b = x * (2.43f * x + 0.59f) + 0.14f;
355 return fmaxf(0.0f, fminf(1.0f, a / b));
356}

358__global__ void tonemap_kernel(const float* __restrict__ linear_fb,
359 unsigned char* __restrict__ argb_out,
360 int width, int height, int total_samples) {
361 int x = blockIdx.x * blockDim.x + threadIdx.x;
362 int y = blockIdx.y * blockDim.y + threadIdx.y;
363 if (x >= width || y >= height) return;
364
365 int px = y * width + x;
366 int tp = width * height;
367 float inv_n = 1.0f / (float)total_samples;
368
369 float r = sqrtf(aces_tonemap_gpu(fmaxf(linear_fb[px] * inv_n, 0.0f)));
370 float g = sqrtf(aces_tonemap_gpu(fmaxf(linear_fb[px + tp] * inv_n, 0.0f)));
371 float b = sqrtf(aces_tonemap_gpu(fmaxf(linear_fb[px + tp*2] * inv_n, 0.0f)));
372
373 int idx = px * 4;
374 argb_out[idx + 0] = (unsigned char)(fminf(b, 1.0f) * 255.0f);
375 argb_out[idx + 1] = (unsigned char)(fminf(g, 1.0f) * 255.0f);
376 argb_out[idx + 2] = (unsigned char)(fminf(r, 1.0f) * 255.0f);
377 argb_out[idx + 3] = 255;
378}

391// ============================================================
392// GPU Denoise kernel : filtre bilateral 3x3 sur le framebuffer lineaire
393// Lisse le bruit en preservant les contours (difference de couleur)
394// ============================================================
395__global__ void denoise_bilateral_kernel(
396 const float* __restrict__ input,
397 float* __restrict__ output,
398 int width, int height, float sigma_color)
399{
400 int x = blockIdx.x * blockDim.x + threadIdx.x;
401 int y = blockIdx.y * blockDim.y + threadIdx.y;
402 if (x >= width || y >= height) return;
403
404 int tp = width * height;
405 int px = y * width + x;
406 float cr = input[px], cg = input[px+tp], cb = input[px+tp*2];
407
408 float sum_r = 0, sum_g = 0, sum_b = 0, wsum = 0;
409 float inv_sigma2 = 1.0f / (2.0f * sigma_color * sigma_color);
410
411 for (int dy = -1; dy <= 1; dy++) {
412 int ny = y + dy;
413 if (ny < 0 || ny >= height) continue;
414 for (int dx = -1; dx <= 1; dx++) {
415 int nx = x + dx;
416 if (nx < 0 || nx >= width) continue;
417 int npx = ny * width + nx;
418 float nr = input[npx], ng = input[npx+tp], nb = input[npx+tp*2];
419 float dr = cr-nr, dg = cg-ng, db = cb-nb;
420 float color_dist = dr*dr + dg*dg + db*db;
421 float w = expf(-color_dist * inv_sigma2);
422 sum_r += nr * w; sum_g += ng * w; sum_b += nb * w;
423 wsum += w;
424 }
425 }
426 float inv_w = 1.0f / (wsum + 1e-8f);
427 output[px] = sum_r * inv_w;
428 output[px+tp] = sum_g * inv_w;
429 output[px+tp*2] = sum_b * inv_w;
430}

470void GpuRenderer::upload_scene(const GSphere* spheres, int num_spheres,
471 const GPlane* planes, int num_planes,
472 const GMaterial* materials, int num_materials,
473 const GBVHNode* bvh_nodes, int num_bvh_nodes) {

492 CUDA_CHECK(cudaMalloc(&d_soa_cy_, num_spheres * sizeof(float)));
493 CUDA_CHECK(cudaMalloc(&d_soa_cz_, num_spheres * sizeof(float)));

503 CUDA_CHECK(cudaMalloc(&d_planes_, num_planes * sizeof(GPlane)));
504 CUDA_CHECK(cudaMemcpy(d_planes_, planes, num_planes * sizeof(GPlane), cudaMemcpyHostToDevice));
505 }
506 if (num_materials > 0) {
507 CUDA_CHECK(cudaMalloc(&d_materials_, num_materials * sizeof(GMaterial)));
508 CUDA_CHECK(cudaMemcpy(d_materials_, materials, num_materials * sizeof(GMaterial), cudaMemcpyHostToDevice));
509 }
510 if (num_bvh_nodes > 0) {
511 CUDA_CHECK(cudaMalloc(&d_bvh_nodes_, num_bvh_nodes * sizeof(GBVHNode)));

514 // L2 cache persistence pour BVH (Ampere CC 8.6+)
515 // Les top nodes sont accedes par quasi tous les rayons
516 size_t bvh_bytes = num_bvh_nodes * sizeof(GBVHNode);
517 cudaDeviceSetLimit(cudaLimitPersistingL2CacheSize, bvh_bytes);

521void GpuRenderer::render(const GCamera& camera, float* h_output, int width, int height) {
522 ensure_framebuffer(width, height);
523
524 // Persistent stream for kernel/transfer overlap
525 static cudaStream_t s_stream = nullptr;
526 if (!s_stream) {
527 cudaStreamCreate(&s_stream);
528
529 // L2 cache persistence : BVH nodes en persistent L2 (Ampere CC 8.6+)
530 if (d_bvh_nodes_ && num_bvh_nodes_ > 0) {
531 cudaStreamAttrValue attr = {};
532 attr.accessPolicyWindow.base_ptr = d_bvh_nodes_;
533 attr.accessPolicyWindow.num_bytes = num_bvh_nodes_ * sizeof(GBVHNode);
534 attr.accessPolicyWindow.hitRatio = 1.0f;
535 attr.accessPolicyWindow.hitProp = cudaAccessPropertyPersisting;
536 attr.accessPolicyWindow.missProp = cudaAccessPropertyStreaming;
537 cudaStreamSetAttribute(s_stream, cudaStreamAttributeAccessPolicyWindow, &attr);
538 }
539 }
540
541 // Upload camera en constant memory
542 cudaMemcpyToSymbol(d_camera_bytes, &camera, sizeof(GCamera));
543
544 // Block size via occupancy API
545 int flat_block_size = 0, min_grid = 0;
546 cudaOccupancyMaxPotentialBlockSize(&min_grid, &flat_block_size, render_kernel, 0, 0);
547 int bx = 16, by = flat_block_size / bx;
548 if (by < 1) by = 1;
549 if (by > 16) by = 16;
550 dim3 block_size(bx, by);
551 dim3 grid_size((width + block_size.x - 1) / block_size.x,
552 (height + block_size.y - 1) / block_size.y);
553
554 cudaEvent_t start, stop;
555 cudaEventCreate(&start);
556 cudaEventCreate(&stop);
557
558 cudaEventRecord(start, s_stream);
559
560 render_kernel<<<grid_size, block_size, 0, s_stream>>>(
561 d_framebuffer_, width, height,
562 d_bvh_nodes_, num_bvh_nodes_,
563 d_spheres_, num_spheres_,
564 d_planes_, num_planes_,
565 d_materials_,
566 d_soa_cx_, d_soa_cy_, d_soa_cz_, d_soa_r_, d_soa_mat_
567 );
568
569 cudaError_t err = cudaGetLastError();
570 if (err != cudaSuccess) {
571 fprintf(stderr, "Kernel error: %s\n", cudaGetErrorString(err));
572 cudaEventDestroy(start); cudaEventDestroy(stop);
573 return;
574 }
575
576 // Transfert async D->H sur le meme stream (sequentialise apres le kernel)
577 // h_output DOIT etre en memoire pinned (cudaMallocHost) pour l'async
578 cudaMemcpyAsync(h_output, d_framebuffer_, width * height * 3 * sizeof(float),
579 cudaMemcpyDeviceToHost, s_stream);
580
581 cudaEventRecord(stop, s_stream);
582 // Synchroniser le stream (attendre kernel + transfert)
583 cudaStreamSynchronize(s_stream);
584
585 cudaEventElapsedTime(&last_ms_, start, stop);
586 cudaEventDestroy(start);
587 cudaEventDestroy(stop);
588}

590// ============================================================
591// Bridge: conversion CPU Scene -> GPU + rendu
592// ============================================================
593static void convert_scene(const Scene& scene,
594 std::vector<GSphere>& out_spheres,
595 std::vector<GPlane>& out_planes,
596 std::vector<GMaterial>& out_materials,
597 GCamera& out_camera) {
598 std::vector<const Material*> mat_ptrs;
599 auto get_mat = [&](const Material* m) -> int {
600 for (int i = 0; i < (int)mat_ptrs.size(); i++)
601 if (mat_ptrs[i] == m) return i;
602 int idx = (int)mat_ptrs.size();
603 mat_ptrs.push_back(m);
604 GMaterial gm;
605 switch (m->type()) {
606 case Material::Type::Lambertian: {
607 auto* lam = static_cast<const Lambertian*>(m);
608 gm.type = GMAT_LAMBERTIAN;
609 gm.albedo = GVec3((float)lam->albedo().x(), (float)lam->albedo().y(), (float)lam->albedo().z());
610 break;
611 }
612 case Material::Type::Metal: {
613 auto* met = static_cast<const Metal*>(m);
614 gm.type = GMAT_METAL;
615 gm.albedo = GVec3((float)met->albedo().x(), (float)met->albedo().y(), (float)met->albedo().z());
616 gm.param = (float)met->fuzz();
617 break;
618 }
619 case Material::Type::Dielectric: {
620 auto* die = static_cast<const Dielectric*>(m);
621 gm.type = GMAT_DIELECTRIC;
622 gm.albedo = GVec3(1, 1, 1);
623 gm.param = (float)die->refraction_index();
624 break;
625 }
626 case Material::Type::Emissive: {
627 auto* em = static_cast<const Emissive*>(m);
628 gm.type = GMAT_EMISSIVE;
629 gm.albedo = GVec3((float)em->emit_color().x(), (float)em->emit_color().y(), (float)em->emit_color().z());
630 gm.param = (float)em->intensity();
631 break;
632 }
633 default: gm.type = GMAT_LAMBERTIAN; gm.albedo = GVec3(0.5f, 0.5f, 0.5f); break;
634 }
635 out_materials.push_back(gm);
636 return idx;
637 };
638
639 for (const auto& obj : scene.world.objects()) {
640 if (auto* sp = dynamic_cast<const Sphere*>(obj.get())) {
641 GSphere gs;
642 gs.center = GVec3((float)sp->center().x(), (float)sp->center().y(), (float)sp->center().z());
643 gs.radius = (float)sp->radius();
644 HitRecord rec;
645 Ray test_ray(sp->center() + Vec3(0, sp->radius() + 1, 0), Vec3(0, -1, 0));
646 gs.material_id = (sp->hit(test_ray, Interval(0.001, 1000.0), rec) && rec.material)
647 ? get_mat(rec.material) : 0;
648 out_spheres.push_back(gs);
649 } else if (auto* pl = dynamic_cast<const Plane*>(obj.get())) {
650 GPlane gp;
651 HitRecord rec;
652 Ray test_ray(Point3(0, 100, 0), Vec3(0, -1, 0));
653 if (pl->hit(test_ray, Interval(0.001, 10000.0), rec) && rec.material) {
654 gp.point = GVec3((float)rec.p.x(), (float)rec.p.y(), (float)rec.p.z());
655 gp.normal = GVec3((float)rec.normal.x(), (float)rec.normal.y(), (float)rec.normal.z());
656 gp.material_id = get_mat(rec.material);
657 }
658 out_planes.push_back(gp);
659 } else {
660 static bool warned = false;
661 if (!warned) {
662 fprintf(stderr, "Warning: GPU ne supporte que Sphere/Plane. Triangles/Cylindres ignores.\n");
663 fprintf(stderr, " Utilisez --scene showcase, final, ou three_spheres pour le GPU.\n");
664 warned = true;
665 }
666 }
667 }
668
669 const Camera& cam = scene.camera;
670 int w = cam.get_image_width(), h = cam.get_image_height();
671 double theta = cam.vfov * 3.14159265358979323846 / 180.0;
672 double vp_h = 2.0 * std::tan(theta / 2.0) * cam.focus_dist;
673 double vp_w = vp_h * ((double)w / h);
674 Vec3 wd = unit_vector(cam.lookfrom - cam.lookat);
675 Vec3 ud = unit_vector(cross(cam.vup, wd));
676 Vec3 vd = cross(wd, ud);
677 Vec3 vu = vp_w * ud, vv = vp_h * (-vd);
678 Vec3 pdu = vu / (double)w, pdv = vv / (double)h;
679 Point3 p00 = cam.lookfrom - cam.focus_dist * wd - vu/2.0 - vv/2.0 + 0.5*(pdu+pdv);
680 double dr = cam.focus_dist * std::tan((cam.defocus_angle/2.0)*3.14159265358979323846/180.0);
681
682 out_camera.origin = GVec3((float)cam.lookfrom.x(), (float)cam.lookfrom.y(), (float)cam.lookfrom.z());
683 out_camera.pixel00_loc = GVec3((float)p00.x(), (float)p00.y(), (float)p00.z());
684 out_camera.pixel_delta_u = GVec3((float)pdu.x(), (float)pdu.y(), (float)pdu.z());
685 out_camera.pixel_delta_v = GVec3((float)pdv.x(), (float)pdv.y(), (float)pdv.z());
686 out_camera.defocus_angle = (float)cam.defocus_angle;
687 out_camera.defocus_disk_u = GVec3((float)(ud.x()*dr), (float)(ud.y()*dr), (float)(ud.z()*dr));
688 out_camera.defocus_disk_v = GVec3((float)(vd.x()*dr), (float)(vd.y()*dr), (float)(vd.z()*dr));
689 out_camera.image_width = w; out_camera.image_height = h;
690 out_camera.samples_per_pixel = cam.samples_per_pixel;
691 out_camera.max_depth = cam.max_depth;
692}

720// ============================================================
721// Simple GPU BVH builder (on spheres only, planes are tested brute-force in leaves)
722// ============================================================
723struct GBuildInfo { int sphere_idx; GAABB bbox; GVec3 centroid; };
724
725static void build_gpu_bvh(std::vector<GBuildInfo>& info, int start, int end,
726 std::vector<GBVHNode>& nodes, int num_planes) {
727 int my_idx = (int)nodes.size();
728 nodes.push_back(GBVHNode());
729
730 GAABB bounds;
731 for (int i = start; i < end; i++) {
732 bounds.mn.x = fminf(bounds.mn.x, info[i].bbox.mn.x);
733 bounds.mn.y = fminf(bounds.mn.y, info[i].bbox.mn.y);
734 bounds.mn.z = fminf(bounds.mn.z, info[i].bbox.mn.z);
735 bounds.mx.x = fmaxf(bounds.mx.x, info[i].bbox.mx.x);
736 bounds.mx.y = fmaxf(bounds.mx.y, info[i].bbox.mx.y);
737 bounds.mx.z = fmaxf(bounds.mx.z, info[i].bbox.mx.z);
738 }
739 nodes[my_idx].bbox = bounds;
740
741 int n = end - start;
742 if (n <= 4) {
743 // Leaf : stores sphere indices [start, end)
744 nodes[my_idx].data = (uint32_t)start;
745 nodes[my_idx].num_prims = (uint16_t)n;
746 nodes[my_idx].split_axis = 0;
747 nodes[my_idx].prim_type = 0; // spheres
748 return;
749 }
750
751 // Find longest axis of centroid bounds
752 GVec3 cmin(1e30f,1e30f,1e30f), cmax(-1e30f,-1e30f,-1e30f);
753 for (int i = start; i < end; i++) {
754 cmin.x = fminf(cmin.x, info[i].centroid.x);
755 cmin.y = fminf(cmin.y, info[i].centroid.y);
756 cmin.z = fminf(cmin.z, info[i].centroid.z);
757 cmax.x = fmaxf(cmax.x, info[i].centroid.x);
758 cmax.y = fmaxf(cmax.y, info[i].centroid.y);
759 cmax.z = fmaxf(cmax.z, info[i].centroid.z);
760 }
761 float dx = cmax.x-cmin.x, dy = cmax.y-cmin.y, dz = cmax.z-cmin.z;
762 int axis = (dx > dy && dx > dz) ? 0 : (dy > dz) ? 1 : 2;
763
764 // Sort by centroid on axis
765 std::sort(info.begin()+start, info.begin()+end,
766 [axis](const GBuildInfo& a, const GBuildInfo& b) {
767 float va = (axis==0)?a.centroid.x:(axis==1)?a.centroid.y:a.centroid.z;
768 float vb = (axis==0)?b.centroid.x:(axis==1)?b.centroid.y:b.centroid.z;
769 return va < vb;
770 });
771
772 int mid = start + n / 2;
773
774 nodes[my_idx].split_axis = (uint8_t)axis;
775 nodes[my_idx].num_prims = 0; // interior
776 nodes[my_idx].prim_type = 0;
777
778 // Left child (immediately after)
779 build_gpu_bvh(info, start, mid, nodes, num_planes);
780 // Right child
781 nodes[my_idx].data = (uint32_t)nodes.size();
782 build_gpu_bvh(info, mid, end, nodes, num_planes);
783}

788void gpu_render_scene(const Scene& scene, ImageBuffer& image, float& kernel_ms) {

790 // Renderer global pour acces depuis gpu_realtime_render_frame

792 static float* s_fb = nullptr;

796
797 // Re-upload scene seulement si elle a change
798 if (!s_renderer || s_last_scene != &scene) {
799 std::vector<GSphere> spheres;
800 std::vector<GPlane> planes;
801 std::vector<GMaterial> materials;
802 convert_scene(scene, spheres, planes, materials, gcam);
803
804 std::vector<GBVHNode> bvh_nodes;
805 if (spheres.size() > 8) {
806 std::vector<GBuildInfo> info(spheres.size());
807 for (int i = 0; i < (int)spheres.size(); i++) {
808 info[i].sphere_idx = i;
809 float r = spheres[i].radius;
810 GVec3 c = spheres[i].center;
811 info[i].bbox = GAABB(GVec3(c.x-r, c.y-r, c.z-r), GVec3(c.x+r, c.y+r, c.z+r));
812 info[i].centroid = c;
813 }
814 build_gpu_bvh(info, 0, (int)info.size(), bvh_nodes, (int)planes.size());
815 std::vector<GSphere> ordered(spheres.size());
816 for (int i = 0; i < (int)info.size(); i++)
817 ordered[i] = spheres[info[i].sphere_idx];
818 spheres = std::move(ordered);
819 fprintf(stderr, "GPU BVH: %d nodes for %d spheres\n", (int)bvh_nodes.size(), (int)spheres.size());
820 }
821
822 delete s_renderer;
823 s_renderer = new GpuRenderer();
824 s_renderer->upload_scene(spheres.data(), (int)spheres.size(),
825 planes.data(), (int)planes.size(),
826 materials.data(), (int)materials.size(),
827 bvh_nodes.empty() ? nullptr : bvh_nodes.data(),
828 (int)bvh_nodes.size());
829 s_last_scene = &scene;
830 } else {
831 // Scene deja uploadee : recalculer seulement la camera
832 convert_camera_only(scene.camera, gcam);
833 }
834
835 int w = gcam.image_width, h = gcam.image_height;
836
837 // Framebuffer pinned persistant
838 if (!s_fb || s_fb_w != w || s_fb_h != h) {
839 if (s_fb) cudaFreeHost(s_fb);
840 cudaMallocHost(&s_fb, w * h * 3 * sizeof(float));
841 s_fb_w = w; s_fb_h = h;
842 }
843
844 s_renderer->render(gcam, s_fb, w, h);
845 kernel_ms = s_renderer->last_render_ms();
846
847 // Copie rapide vers ImageBuffer
848 int total_px = w * h;
849 for (int y = 0; y < h; y++)
850 for (int x = 0; x < w; x++) {
851 int px = y * w + x;
852 image.set_pixel(x, y, Color(
853 fminf(s_fb[px], 1.0f),
854 fminf(s_fb[px + total_px], 1.0f),
855 fminf(s_fb[px + total_px*2], 1.0f)));
856 }
857}

942// Cached per-session (pas recalcule a chaque frame)
943static dim3 s_rt_block, s_rt_grid;
944static int s_rt_acc_grid = 0;
945static cudaEvent_t s_rt_start = nullptr, s_rt_stop = nullptr;
946static bool s_rt_config_done = false;
947
948float gpu_realtime_render_frame(const Scene& scene) {
949 GCamera gcam;
950 convert_camera_only(scene.camera, gcam);
951 cudaMemcpyToSymbol(d_camera_bytes, &gcam, sizeof(GCamera));
952
953 extern GpuRenderer* s_renderer;
954 if (!s_renderer) return 0;
955
956 // Config une seule fois (block size, grid, events)
957 if (!s_rt_config_done) {
958 s_renderer->ensure_framebuffer(s_rt_w, s_rt_h);
959 int flat_block = 0, min_grid = 0;
960 cudaOccupancyMaxPotentialBlockSize(&min_grid, &flat_block, render_kernel, 0, 0);
961 int bx = 16, by = flat_block / bx;
962 if (by < 1) by = 1; if (by > 16) by = 16;
963 s_rt_block = dim3(bx, by);
964 s_rt_grid = dim3((s_rt_w + bx - 1) / bx, (s_rt_h + by - 1) / by);
965 int total_elems = s_rt_w * s_rt_h * 3;
966 s_rt_acc_grid = (total_elems + 255) / 256;
967 cudaEventCreate(&s_rt_start);
968 cudaEventCreate(&s_rt_stop);
969 s_rt_config_done = true;
970 }
971
972 cudaEventRecord(s_rt_start, s_rt_stream);
973
974 // Render + accumulate, tout sur le stream (pas de sync entre les deux)
975 render_kernel<<<s_rt_grid, s_rt_block, 0, s_rt_stream>>>(
976 s_rt_frame_fb, s_rt_w, s_rt_h,
977 s_renderer->d_bvh_nodes_, s_renderer->num_bvh_nodes_,
978 s_renderer->d_spheres_, s_renderer->num_spheres_,
979 s_renderer->d_planes_, s_renderer->num_planes_,
980 s_renderer->d_materials_,
981 s_renderer->d_soa_cx_, s_renderer->d_soa_cy_,
982 s_renderer->d_soa_cz_, s_renderer->d_soa_r_, s_renderer->d_soa_mat_);
983
984 accumulate_kernel<<<s_rt_acc_grid, 256, 0, s_rt_stream>>>(
985 s_rt_frame_fb, s_rt_accum_fb, s_rt_w * s_rt_h);
986
987 cudaEventRecord(s_rt_stop, s_rt_stream);
988 // PAS de cudaStreamSynchronize ici - le tonemap synchronisera
989 s_rt_accum_count++;
990
991 float ms = 0;
992 cudaEventSynchronize(s_rt_stop);
993 cudaEventElapsedTime(&ms, s_rt_start, s_rt_stop);
994 return ms;
995}

997int gpu_realtime_tonemap(unsigned char* argb_host, int width, int height) {
998 dim3 block(16, 16);
999 dim3 grid((width + 15) / 16, (height + 15) / 16);
1000
1001 tonemap_kernel<<<grid, block, 0, s_rt_stream>>>(
1002 s_rt_accum_fb, s_rt_argb, width, height, s_rt_accum_count);
