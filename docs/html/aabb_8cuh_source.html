25 // Construction depuis deux coins opposes (ordre quelconque)
26 RT_HOSTDEV AABB(const Point3& a, const Point3& b) {
27 x = (a[0] <= b[0]) ? Interval(a[0], b[0]) : Interval(b[0], a[0]);
28 y = (a[1] <= b[1]) ? Interval(a[1], b[1]) : Interval(b[1], a[1]);
29 z = (a[2] <= b[2]) ? Interval(a[2], b[2]) : Interval(b[2], a[2]);
30 }

43 // Test slab de Williams et al. : intersection rayon-boite en 6 mul + 6 cmp.
44 // inv_dir precalcule dans Ray transforme (bound - origin) / dir en multiplication.
45 // On resserre progressivement [t_min, t_max] par axe ; si l'intervalle
46 // devient vide (t_max < t_min), le rayon rate la boite -> early exit.
47 RT_HOSTDEV bool hit(const Ray& r, double t_min, double t_max) const {
48 const Point3& ro = r.orig;
49 const Vec3& inv = r.inv_dir;
50
51 // Axe X
52 double tx1 = (x.min - ro[0]) * inv[0];
53 double tx2 = (x.max - ro[0]) * inv[0];
54 if (tx1 > tx2) { double tmp = tx1; tx1 = tx2; tx2 = tmp; }
55 if (tx1 > t_min) t_min = tx1;
56 if (tx2 < t_max) t_max = tx2;
57 if (t_max < t_min) return false;
58
59 // Axe Y
60 double ty1 = (y.min - ro[1]) * inv[1];
61 double ty2 = (y.max - ro[1]) * inv[1];
62 if (ty1 > ty2) { double tmp = ty1; ty1 = ty2; ty2 = tmp; }
63 if (ty1 > t_min) t_min = ty1;
64 if (ty2 < t_max) t_max = ty2;
65 if (t_max < t_min) return false;
66
67 // Axe Z
68 double tz1 = (z.min - ro[2]) * inv[2];
69 double tz2 = (z.max - ro[2]) * inv[2];
70 if (tz1 > tz2) { double tmp = tz1; tz1 = tz2; tz2 = tmp; }
71 if (tz1 > t_min) t_min = tz1;
72 if (tz2 < t_max) t_max = tz2;
73 if (t_max < t_min) return false;
74
75 return true;
76 }

78 // Surcharge pour les tests qui passent un Interval
79 RT_HOSTDEV bool hit(const Ray& r, Interval ray_t) const {
80 return hit(r, ray_t.min, ray_t.max);
81 }

83 // Surface totale de la boite : utilisee par la SAH (Surface Area Heuristic)
84 // pour estimer le cout de split lors de la construction du BVH.
85 // Cout ~ surface_enfant / surface_parent * nb_primitives_enfant
86 RT_HOSTDEV double surface_area() const {
87 double dx = x.size(), dy = y.size(), dz = z.size();
88 return 2.0 * (dx*dy + dy*dz + dz*dx);
89 }

91 // Axe le plus long : heuristique de split quand SAH n'est pas utilisee
92 RT_HOSTDEV int longest_axis() const {
93 double sx = x.size(), sy = y.size(), sz = z.size();
94 if (sx > sy) return (sx > sz) ? 0 : 2;
95 return (sy > sz) ? 1 : 2;
96 }
